def create
  @artifact = Artifact.new(artifact_params)

  respond_to do |format|
    if @artifact.save
      format.html {redirect_to @artifact, notice: 'Artifact was successfully created.'}
      format.json {render :show, status: :created, location: @artifact}
    else
      format.html {render :new}
      format.json {render json: @artifact.errors, status: :unprocessable_entity}
    end
  end
end

def destroy
  if @artifact.user == current_user || current_user.is_admin == true
    if @artifact.projects.empty?
      @artifact.destroy
      respond_to do |format|
        format.html {redirect_to artifacts_url, notice: 'Artifact was successfully destroyed.'}
        format.json {head :no_content}
      end
    end
  end
end

def index
  @artifacts = Artifact.all
  if params[:search]
    @artifacts = Artifact.where('name LIKE ? OR description LIKE ?',
                                "%#{params[:search]}%",
                                "%#{params[:search]}%").order(created_at: :desc)
  else
    @artifacts = Artifact.all
  end
end
