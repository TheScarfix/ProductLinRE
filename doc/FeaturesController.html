def create
  @project = Project.find(params[:project_id])
  @feature = @project.features.create(feature_params)
  respond_to do |format|
    if @feature.save
      format.html {redirect_to editor_index_url(project: @project.id), notice: 'Feature was successfully created.'}
      format.json {render :show, status: :created, location: @feature}
    else
      format.html {render :new}
      format.json {render json: @feature.errors, status: :unprocessable_entity}
    end
  end
end

def destroy
  if @feature.project.user == current_user || current_user.is_admin == true
    project_id = @feature.project_id
    @feature.destroy
    respond_to do |format|
      format.html {redirect_to editor_index_url(project: project_id.to_s), notice: 'Feature was successfully destroyed.'}
      format.json {head :no_content}
    end
  end
end

def edit
  artifact_ids = []

  @feature.artifacts.each do |artifact|
    artifact_ids << artifact.id
  end
  passage_ids = []

  @feature.passages.each do |passage|
    passage_ids << passage.id
  end

  @all_artifacts = Artifact.order(updated_at: :desc).where.not(id: artifact_ids)
  @all_passages = Passage.order(updated_at: :desc).where.not(id: passage_ids)
end

def update
  if @feature.project.user == current_user
    if feature_params[:to_remove] != ""
      to_remove_array = feature_params[:to_remove].split(";")
      to_remove_array.each do |to_remove_value|
        if to_remove_value[0] == 'a'
          arti_id = to_remove_value.split("_")[1]
          @feature.artifacts.delete(Artifact.find(arti_id))
        end
        if to_remove_value[0] == 'p'
          pass_id = to_remove_value.split("_")[1]
          @feature.passages.delete(Passage.find(pass_id))
        end
      end
    end
    if feature_params[:to_add] != ""
      to_remove_array = feature_params[:to_add].split(";")
      to_remove_array.each do |to_remove_value|
        if to_remove_value[0] == 'a'
          arti_id = to_remove_value.split("_")[1]
          @feature.artifacts << Artifact.find(arti_id)
        end
        if to_remove_value[0] == 'p'
          pass_id = to_remove_value.split("_")[1]
          @feature.passages << Passage.find(pass_id)
        end
      end
    end
    respond_to do |format|
      if @feature.update(feature_params)
        format.html {redirect_to @feature, notice: 'Feature was successfully updated.'}
        format.json {render :show, status: :ok, location: @feature}
      else
        format.html {render :edit}
        format.json {render json: @feature.errors, status: :unprocessable_entity}
      end
    end
  end
end
