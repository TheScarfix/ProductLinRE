def index
  #list own projects in home
  @own_projects = Project.where(user_id: current_user.id).all

  #list own projects in left sidebar
  @own_last_projects = @own_projects.order(updated_at: :desc).limit(10)

  #list the 10 last projects from other users in leftsidbar and home
  @other_projects = Project.where.not(user_id: current_user.id).order(updated_at: :desc).limit(10)

  #list own project's features in features overview
  @current_project = Project.find_by_id(params[:project])

  if params[:project]
    session[:cur_project_id] = params[:project]
  end
end
