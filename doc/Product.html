def compare(difference)
  @product.features.each do |feature|
    unless @product.orig_product.features.include? feature
      difference.missing_features.append(feature)
    end
  end
  @product.orig_product.features.each do |feature|
    unless @product.features.include? feature
      difference.deleted_features.append(feature)
    end
  end
  @product.features.each do |feature|
    if @product.orig_product.features.include? feature
      @product.orig_product.find(feature).compare(difference)
    end
  end
end

def download
  zip_file_path = Rails.root.join("tmp/tempfiles/product_#{id}.zip")

  # see if the file exists already, and if it does, delete it.
  if File.file?(zip_file_path)
    File.delete(zip_file_path)
  end

  # open or create the zip file
  Zip::OutputStream.open(zip_file_path) do |zipfile|
    features.each do |feature|
      feature.artifacts.each do |artifact|
        zipfile.put_next_entry(artifact.file.filename)
        zipfile.write artifact.file.download
      end
      feature.passages.each do |passage|
        zipfile.put_next_entry(passage.file.filename)
        zipfile.write passage.file.download
      end
    end
    zipfile.close
  end
  zip_file_path
end
