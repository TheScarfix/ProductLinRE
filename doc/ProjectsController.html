def create
  @project = Project.new(project_params)

  respond_to do |format|
    if @project.save
      format.html {redirect_to editor_index_path(project: @project.id), notice: 'Project was successfully created.'}
      format.json {render :show, status: :created, location: @project}
    else
      format.html {render :new}
      format.json {render json: @project.errors, status: :unprocessable_entity}
    end
  end
end

def destroy
  if @project.user == current_user || current_user.is_admin == true
    @project.destroy
    respond_to do |format|
      format.html {redirect_to editor_index_url, notice: 'Project was successfully destroyed.'}
      format.json {head :no_content}
    end
  end
end

def index
  @projects = current_user.projects
  if params[:search]
    @projects = Project.where("name LIKE ? OR description LIKE ?", "%#{params[:search]}%", "%#{params[:search]}%").order(created_at: :desc)
  else
    @projects = Project.all
  end
end

def update
  if @project.user == current_user
    if project_params[:products_features]
      products_hash = project_params[:products_features].to_hash
      @project.products.each do |product|
        product.update(products_hash[product.id.to_s])
      end
    end
    respond_to do |format|
      if @project.update(project_params.except(:products_features))
        format.html {redirect_to editor_index_path(project: session[:cur_project_id]), notice: 'Project was successfully updated.'}
        format.json {render :show, status: :ok, location: @project}
      else
        format.html {render :edit}
        format.json {render json: @project.errors, status: :unprocessable_entity}
      end
    end
  end
end
